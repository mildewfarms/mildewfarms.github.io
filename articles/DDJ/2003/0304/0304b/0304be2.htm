int ns__trySoap(struct soap * s, char * name, Response * rS)
{
    rS->status = 1;
    char tempStr[250];
    strcpy(tempStr, "\"Hello ");
    if(name == NULL || strlen(name) > 100)
        return SOAP_CLI_FAULT;
    strcat(tempStr,  name);
    strcat (tempStr, "\" says ns__trySoap");
    rS->message = strdup(tempStr);
    return SOAP_OK;
}
