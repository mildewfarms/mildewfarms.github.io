#include <time.h>
#include <fcgi_stdio.h>
 
main()
{
   time_t tod;
   long count = 0;
   while(FCGI_Accept() >= 0) {
      printf("Content-type: text/html\r\n");
      printf("\r\n");
      time(&tod);
      printf("The current time of day is %s <BR>", ctime(&tod));
      printf("\r\n");
      printf("This page has been accessed %d times<BR>", ++count);
   }
   exit(0);
}
