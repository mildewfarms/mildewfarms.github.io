SELECT DISTINCT serialwkg, incnum, duedate, description
   FROM incident i, empserv e
   WHERE i.serialwkg=e.serial
      AND duedate > '1996-9-01'
      AND dept = 'S21'
   GROUP BY serialwkg, incnum, duedate, description
   ORDER BY serialwkg
