void main(void)
{
   jmp_buf jumper;
   int LocalVar = 1;
   printf("1: %i\n", LocalVar);
   if (setjmp(jumper) == 0)
   {
      LocalVar++;
      printf("2: %i\n", LocalVar);
      longjmp(jumper, 1);
   }
   LocalVar++;
   printf("3: %i\n", LocalVar);
}
