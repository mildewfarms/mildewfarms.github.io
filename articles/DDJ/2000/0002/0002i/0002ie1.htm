VAORC VNetInfo::Run(LPVAOV2LSTRUCT pV2L)
{
    CNetworkResourceCollector nrCollector(this);
    BOOL rc = FALSE;
    rc = nrCollector.Run();
    ASSERT(rc);
    if(!(rc))
    {
        AfxMessageBox(IDE_GENERAL_RESOURCE_COLLECTION, MB_OK | MB_ICONSTOP);
    }
    return(VAORC_SUCCESS);
}
