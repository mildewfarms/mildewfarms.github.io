expression      ->  term r-expression
r-expression    ->  epsilon
                    | "+" term r-expression
                    | "-" term r-expression
term            ->  factor r-term
r-term          ->  epsilon
                    | "+" factor r_term
                    | "-" factor r_term
factor          ->  "(" expression ")"
                    | integer
