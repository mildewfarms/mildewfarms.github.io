Becoming

By Barry Boone

Dr. Dobb's Journal Fall 1998 

class A {
   int i = 1;
   void method() {
      System.out.println(i);
   }
}
class B extends A {
   int i = 2;
   void method() {
      System.out.println(i);
   }
}
class Test {
   public static void main(String[] args) {
      A a= new B();
      a.method();
      System.out.println(a.i);
   }
}
