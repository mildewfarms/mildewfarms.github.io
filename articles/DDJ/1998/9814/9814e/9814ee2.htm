class Info {
   int data;
   public String toString() {
      return new Integer(data).toString();
   }
}
(b)
class Test {
   public static void main(String[] args) {
      Info i = new Info();
      i.data = 1;
      change(i);
      System.out.println(i);
   }
   static void change(Info i) {
      i.data *= 2;
      i = null;
   }
}
