Becoming

Dr. Dobb's Journal Fall 1998 

class A {
   int i;
   A(int i) {
      this.i = i;
      System.out.println("1");
   }
   A(int i, int j) {
      this(i);
      this.i = i + j;
      System.out.println("2");
   }
}
class B extends A {
   B(int i, int j) {
      System.out.println("3");
   }
   public static void main(String[] args) {
      B b = new B(1, 2);
   }
}
