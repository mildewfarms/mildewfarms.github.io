int rstrcmp(char *s, char *t)
{   if (*s == *t) {
        if (*s)
            return rstrcmp(s+1, t+1);
        else
            return 0;
    } else
        return (*s - *t);
}
(c)
int rcstrcmp(char *s, char *t)
{   return (*s == *t)
           ? (*s ? rcstrcmp(s+1, t+1) : 0)
           : *s - *t;
}
