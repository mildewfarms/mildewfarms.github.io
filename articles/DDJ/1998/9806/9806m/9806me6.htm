(b)
void icount(Tptr p)
{   Tptr s[MAXN];
    int sp = 0;
    for (;;) {
        while (p) {
            s[sp++] = p;
            p = p->lo;
        }
        if (sp == 0) break;
        p = s[--sp];
        count++;
        p = p->hi;
    }
}
