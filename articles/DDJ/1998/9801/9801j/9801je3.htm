typedef struct _MEMORYSTATUS {

/** @dll.struct() */
class MEMORYSTATUS
{
    public int dwLength;     
    public int dwMemoryLoad; 
    public int dwTotalPhys;  
    public int dwAvailPhys;  
    public int dwTotalPageFile;
    public int dwAvailPageFile;

    public int dwAvailVirtual; 
}
