void AddressBook::newItemSelected( QListViewItem * item )
{
  // update line edits
  if ( item ) {
    firstLineEdit->setText( item->text(0) );
    lastLineEdit->setText( item->text(1) );
    emailLineEdit->setText( item->text(2) );
  }
}
