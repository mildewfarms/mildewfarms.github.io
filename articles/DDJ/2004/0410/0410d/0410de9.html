   virtual void f() abstract;
};

   virtual void f(); // overrides A::f and B::f
};
struct E : A, B {
   virtual void g() = B::f; // overrides B::f only, E is abstract
};
struct F : A, B {
   virtual void f() override; // overrides A::f and B::f
