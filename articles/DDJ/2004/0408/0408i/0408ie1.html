#include "callout.h"

LoadLibrary("callout.dll");
HWND hwndCallout = CreateWindow("Callout",
                    "Guess what this icon does...",
                    WS_POPUP, 0, 0, 0, 0,
                    NULL, 0, hInstance, NULL);
SendMessage(hwndCallout, CM_SETANCHOR, 0, MAKELONG(100, 200));
ShowWindow(hwndCallout, SW_SHOW);
