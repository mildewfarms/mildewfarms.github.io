extern inline int button_state(void) {
	int rv;
__asm__ __volatile__(
	"mov.b @0xb7:8,%0l\n"
	"mov.b @0xbe:8,%0h\n"
	: "=r" (rv)				// output    (general register)
	: 					// input     (none)
	: "cc"				// clobbered (condition codes)
	);
	return rv;
}
