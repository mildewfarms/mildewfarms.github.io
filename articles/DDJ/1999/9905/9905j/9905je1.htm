#include <stdio.h>
int main(int argc, char *argv[])
{
    int i;
    FILE *f;
    /* ... */
    for (i = 2; i < argc; i++)  {
        f = fopen(argv[i], "r");
        if (f != NULL)  {
            /* process f */
            fclose(f);
        }
    }
    return 0;
}
