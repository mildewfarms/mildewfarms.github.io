MotorControlData* MotorControlCtor(void)
{
    MotorControlData* d;

    d = Allocate(sizeof(MotorControlData));
    d->mAccelRate = 0;

    if (InstCount++ == 0)
    {
        CREATE_VTABLE(MotorControl, cMotorControl, MOTOR_CONTROL);
        MotorCtor(d, &MotorControl->InitPort);
    }
    else
    {
        MotorCtor(d, NULL);
    }

    return d;
}
