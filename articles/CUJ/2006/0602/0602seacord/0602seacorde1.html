Example 1: hbAssignCodes() function.

void hbAssignCodes(
  int *code, unsigned char *length,  
  int minLen, int maxLen, int alphaSize )  {  
  int n, vec, i;  
  vec = 0;  
  for (n = minLen; n <= maxLen; n++) {  
    for (i = 0; i < alphaSize; i++)  
      if (length[i] == n) { code[i] = vec; vec++; };  
        vec <<= 1;  
  }  
}
