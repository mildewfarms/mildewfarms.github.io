Example 3: Some C++ features are supported in Ch for interactive execution of C++ code.

> int i 
> cin >> i 
10 
> cout << i 
10 
> class tagc {private: int m_i; public: void set(int); int get(int &);} 
> void tagc::set(int i) {m_i = 2*i;} 
> int tagc::get(int &i) {i++; return m_i;} 
> tagc c 
> c.set(20) 
> c.get(i) 
40 
> i 
11 
> sizeof(tagc) 
4
