Listing 1: The program resulting from next_permutation

#include <iostream>
#include <algorithm>

int main()
{
    int a[] = { 1, 2, 3, 4, 5, 6, 7, 8, 9 };
    int count = 0;
    do
    {
        if ( ( a[0] + a[1] + a[2] ) == 17 &&
             ( a[0] + a[3] + a[6] ) == 17 &&
             ( a[2] + a[5] + a[8] ) == 17 &&
             ( a[2] + a[4] + a[6] ) == 17 &&
             ( a[6] + a[7] + a[8] ) == 17 )
        {
            std::cout << count << " : ";
            for ( int i = 0 ; i < 9 ; i++ )
                std::cout << a[ i ] << " ";
            std::cout << "\n";
        }
        count++;
    } while ( std::next_permutation( a, a + 9 ) );
    std::cout << count << " permutations were tested\n";
    return 0;
}
