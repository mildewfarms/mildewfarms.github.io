[Serializable()] 
public class RemoteMessage
{
  private string strMessageID = "";
  private DataSet dsMessageData = null;   
  
  public RemoteMessage()
  {
      
  }

  public string MessageID
  {
    get
    {
          return (strMessageID);    
    }
    set
    {
          strMessageID = value;
    }
  }

  public DataSet MessageData
  {
    get
    {
        return (dsMessageData);  
    }

    set
    {
        dsMessageData = value;
    }
  }
}
