// Read/write UTF-8 file until EOF or UTF-8 encoding error.
int main()
{
  using namespace std;
  using namespace UTF8Lib;
  UTF8Filestream<> fin ("utf8_test.txt",  ios_base::in );
  UTF8Filestream<> fout("utf8_test1.txt", ios_base::out);
  wchar_t wch;
  while(fin.get(wch)) { fout.put(wch); }

}
