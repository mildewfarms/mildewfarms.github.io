Listing 2: Complete Hello Lambda program.

#include <boost/lambda/lambda.hpp>
int main() {
  using namespace boost::lambda;

  std::vector<std::string> vec;
  vec.push_back("This");
  vec.push_back(" is");
  vec.push_back(" extremely");
  vec.push_back(" cool.");

  std::for_each(vec.begin(),vec.end(),std::cout << _1);
}
