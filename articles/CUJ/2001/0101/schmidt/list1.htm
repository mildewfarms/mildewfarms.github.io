#include <stdio.h>

class OldBank
    {
public:
    OldBank() : pass(123)
        {
        }
    void dump() const
        {
        printf("OldBank %d\n", pass);
        }
private:
    int pass;
    };

class NewBank
    {
public:
    NewBank()
        {
        pass(123);
        }
private:
    int pass(int value)
        {
        static int pass = value;
        return pass;
        }
    };
