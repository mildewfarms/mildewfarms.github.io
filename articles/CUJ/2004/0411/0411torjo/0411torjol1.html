logon_dlg::logon_dlg() {
    // note: ID_username and ID_passw are loaded from resources
    tip_text(ID_remember, member_f(&logon_dlg::remember_me,this);
    tip_text(IDOK, "Click here to logon");
    tip_text(IDCANCEL, "Close this window");
}
std::string logon_dlg::remember_me() {
  if ( child<check_box>(ID_remember)->is_checked())
    return "Click this to remember you next time.";
  else
    return "We remember you...";
}
