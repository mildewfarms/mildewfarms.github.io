Example 3: ParamIO XML trees.

int main()
{
    ParamIO xml1, xml2;
    xml1.readFile("ColoredText.xml");
    xml2.readFile("ColoredText.xml");
    assert(xml1 == xml2);

    // change the TEXT value in tree 2
    xml2.write("PARAMS:TEXT", "NewText");
    assert(xml1 != xml2);
    assert(xml1.compare(
           xml2, "PARAMS:TEXT") == false);
    assert(xml1.compare(
           xml2, " PARAMS:COLOR") == true);
    return 0;
}
