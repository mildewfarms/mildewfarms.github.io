Figure 10: Building main4 under AIX.


AIX:sample1> gmake main4
gcc -c -o owncode3.o owncode3.cc
ld -o libowncode.o -r owncode3.o /usr/local/lib/libgcc.a
gcc -nostdlib -shared -o libowncode.so -Wl,-bhalt:4 \
           -Wl,-bI:main4.exp -Wl,-bE:libowncode.exp -lc  libowncode.o
xlC -c -o main4.o main.cc
xlC -c -o native_wrap.o native_wrap.cc
xlC -L. main4.o libowncode.so native_wrap.o -L./native_library/AIX -lnative \
    -Wl,-bE:main4.exp -Wl,-binitfini:_GLOBAL__DI:_GLOBAL__DD -o main4 AIX:sample1>
