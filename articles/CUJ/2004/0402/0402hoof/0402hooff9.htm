Figure 9: Building and running main4 under HP-UX.


HP-UX:sample1> gmake main4
aCC -c -o main4.o main.cc
gcc -fPIC -c -o owncode3.o owncode3.cc
gcc -shared -fPIC owncode3.o -o libowncode.sl \
       -Wl,-E -Wl,+e,mainloop -Wl,+e,_GLOBAL__DI -Wl,+e,_GLOBAL__DD
aCC -c -o native_wrap.o native_wrap.cc
aCC main4.o libowncode.sl native_wrap.o -L./native_library/HP-UX -lnative -o main4 \
       -Wl,+init,_GLOBAL__DI -Wl,+fini,_GLOBAL__DD
