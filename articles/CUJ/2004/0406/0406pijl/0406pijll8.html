define pset
  set $settype = &$arg0.t.header->value_field
  set $settypep = &$settype
  setbegin $arg0
  set $setloop = $setbegin
  setend $arg0
  while ($setloop != $setend)
    set *$settypep = $setloop + $cValueOffset
    output **$settypep
    echo \n
    setnext $setloop
    set $setloop = $setnext
  end
end
