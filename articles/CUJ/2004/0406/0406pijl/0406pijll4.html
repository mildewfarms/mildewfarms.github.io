#########################################
define pveci
  output *($arg0.start + $arg1)
  echo \n
end

document pveci
Print the i-th element of a vector
end

#########################################
define pvec
  veclen $arg0
  set $lloop = 0
  while ($lloop < $veclen)
    pveci $arg0 $lloop
    set $lloop = $lloop + 1
  end
end

document pvec
pvec varname
Print all the elements of a vector varname
example:
09  vector<int> l_vec;
10  l_vec.push_back(1);
11  l_vec.push_back(2);
(gdb) pvec l_vec
1
2
end
