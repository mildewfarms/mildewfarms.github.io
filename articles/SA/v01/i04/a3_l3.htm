Listing 3: Automounter maps example

# Distfile for rdist command
HOSTS = ( dasher dancer prancer vixen comet
cupid donner blitzen rudolf )
NIS_FILES = ( /etc/passwd /etc/hosts /etc/group
/etc/aliases /etc/auto.master
/etc/auto.home )
MAP_HOSTS = ( dasher dancer prancer vixen comet
cupid donner blitzen )
nis:
${NIS_FILES} -> ${HOSTS}
install;

maps:
/etc/auto.direct -> ${MAP_HOSTS}
install;

/sysadmin/auto.direct.rudolf -> rudolf
install /etc/auto.direct;
