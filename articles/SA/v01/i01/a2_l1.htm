Listing 1

SYS=`uname`
PS1='$SYS:${PWD} !$ '
export PS1
tty=`tty | sed 's;/dev/;;'`
case $tty in
tty0[1-4])
TERM=ansi;export TERM
/usr/local/colors
;;
tty1A)
LINES=24;export LINES
/usr/lbin/idleout 5 tty1A
;;
ttyp*)
TERM=ansi;export TERM
echo "You are now logged into system" $SYS
/usr/local/colors
;;
esac
