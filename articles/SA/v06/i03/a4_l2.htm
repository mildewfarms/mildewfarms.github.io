#!/bin/sh

#
# bbdef.sh
#
# BIG BROTHER DEFINITIONS AREA
#

#
# THESE THINGS ARE NEEDED FOR TESTING
#
LYNX="$BBHOME/bin/bbnet"          # PROGRAM TO USE FOR http TESTS
KERMIT="/usr/local/bin/kermit"    # WHERE kermit LIVES

#
# WARNING AND PANIC LEVELS FOR DIFFERENT THINGS
# SEASON TO TASTE
#
PAGER=                            # THE PAGER NUMBER TO CALL (YOURS)
DFPAGE=N                          # PAGE ON DISK FULL (Y/N)
CPUPAGE=N                         # PAGE FOR CPU Y/N
TELNETPAGE=N                      # PAGE ON TELNET FAILURE?
TELNETPAGE=N                      # PAGE ON TELNET FAILURE?
HTTPPAGE=N                        # PAGE ON HTTP FAILURE?

DFWARN=90                         # DISK PCT TO WARN (YELLOW)
DFPANIC=95                        # DISK PCT TO PANIC (RED)
CPUPANIC=300                      # PANIC (RED) AT LOAD AVG OF 3
CPUWARN=150                       # WARN (YELLOW) AT LOAD OF 1.5

PROCS="http sendmail bbd"         # THESE PROCESSES MUST EXIST
PAGEPROC="http"                   # PAGE IF ONE OF THESE IS TOAST
MSGS="NOTICE WARNING"             # MESSAGES TO LOOK FOR (WARNING)
PAGEMSG="NOTICE"                  # PAGE ON THIS ONE

. ${BBHOME}/etc/bbsys.sh          # INCLUDE SYSTEM DEFINITIONS
. ${BBHOME}/etc/bbinc.sh          # INCLUDE BB DEFINITIONS
