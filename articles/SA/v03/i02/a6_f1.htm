#
#   Hayes Smartmodem --
#
hayes   =,-,    "" \dA\pTE1V1X1Q0S2=255S12=255\r\c OK\r \EATDT\T\r\c CONNECT

#
# Pager Dialer Protocol
#
# NOTE : Basically, this is the Hayes modem protocol, with no more
#        interaction after the number has been dialed (\EATDT\T)
#
pager   =,-,    "" \dA\pTE1V1X1Q0S2=255S12=255\r\c OK\r \EATDT\T
