Listing 3: Stbtape.sh--The start logical logging script

# Stbtape.sh - Start Logical Logging
INFORMIXDIR=/usr/informix;  export INFORMIXDIR
NOHUP=/bin/nohup;           export NOHUP
RM=/bin/rm;                 export RM
CD=/bin/cd;                 export CD
PWD=/usr/ed/arch;           export PWD
$CD $INFORMIXDIR
$RM nohup.out
$NOHUP $INFORMIXDIR/bin/tbtape -c < $PWD/tbtape_cr
1>/dev/null 2>&1 &
