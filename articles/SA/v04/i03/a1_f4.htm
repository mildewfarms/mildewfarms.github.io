LOCAL_NET_CONFIG
# Relay everything DNS pukes on, and that appears to be in our domain,
# to the domain address itself.
# Valid domain addresses are handled the normal way
# Throw everything else DNS pukes on over the wall to UUCP_RELAY
# pass to name server to make hostname canonical
R$* < @ $+ . $D . > $*  $#smtp $@ $2.$D $: $1 < @ $2.$D > $3
R$* < @ $D . > $*       $#smtp $@ $D $: $1 < @ $D > $2
R$* < @ $* $~P > $*     $: $1 < @ $[ $2 $3 $: $2 $3.NOTFOUND $] > $4
R$+ < @ $+.$D.NOTFOUND > $*     $#local $: $1
R$+ < @ $*.NOTFOUND > $*        $#uucp $@ sun4nl $: $1 < @ $2 > $3
R$+ < @ $* > $* $#smtp $@ $2 $: $1 < @ $2 > $3
