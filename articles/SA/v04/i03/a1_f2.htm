#####  @(#)local.m4     8.6 (Berkeley) 10/24/93  #####

Mlocal, P=/usr/local/bin/procmail, F=lsDFMShPfn, S=10, R=20/40,
A=procmail -a $h -d $u
Mprog,  P=/bin/sh, F=lsDFMeu, S=10, R=20/40, D=$z:/,
