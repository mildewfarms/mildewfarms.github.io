Listing 1: The pppconnect script

#!/bin/sh

DEFAULTDEVICE=/dev/ttyS2
STATUSFILE=/etc/ppp/mirrorring

/bin/rm -f $STATUSFILE

if [ "x$1" = "x" ]
then
DEVICE=$DEFAULTDEVICE
else
DEVICE="/dev/$1"
fi

pppd connect "chat -f /etc/ppp/pppchat" $DEVICE 38400 noipdefault ipcp-
accept-local ipcp-accept-remote defaultroute :
