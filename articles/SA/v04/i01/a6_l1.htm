# @(#)openwin-menu      23.15 91/09/14 openwin-menu

#       OpenWindows default root menu file - top level menu
#
#       Customized: Lawrence S Reznick, 94Oct12

"Programs" MENU         $HOME/ow/ow-menu-prog

"Utilities" MENU        $HOME/ow/ow-menu-util

SEPARATOR
