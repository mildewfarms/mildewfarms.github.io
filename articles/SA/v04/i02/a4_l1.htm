#!/bin/ksh
# written by Steve Peterson, 11-94.
if (($# != 1))
then print "Usage: getattr devname"; exit; fi
lsattr -El $1 >/tmp/t1.$$
if (($? != 0)); then exit; fi
lsattr -Dl $1 >/tmp/t2.$$
