Figure 1: Using cp with update and backup options

marty@compaq$ ls
marty@compaq$ touch foo; sleep 2; touch bar; ls -l --full-time
total 0
-rw-r--r-- 1 marty    users    0 Tue Sep 27 08:57:23 1994 bar
-rw-r--r-- 1 marty    users    0 Tue Sep 27 08:57:20 1994 foo
marty@compaq$ cp --verbose --version_control=numbered --update --backup foo bar
marty@compaq$ cp --verbose --version_control=numbered --update --backup foo bar
bar -> foo
marty@compaq$ ls -l --full-time
total 0
-rw-r--r-- 1 marty    users    0 Tue Sep 27 08:57:23 1994 bar
-rw-r--r-- 1 marty    users    0 Tue Sep 27 08:57:48 1994 foo
-rw-r--r-- 1 marty    users    0 Tue Sep 27 08:57:20 1994 foo.~1~
