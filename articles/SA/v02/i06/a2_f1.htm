Figure 1: Sample /etc/printcap with local and remote entry for lp2

#
# The network entry for lp2
#

lp2|donald postscript printer:\
:lp=:\
:rm=donald:\
:rp=donald-lp2:\
:sd=/var/spool/lpd/lp2:\
:lf=/var/log/lpd-errs

#
# The host entry for lp2 on donald
#

donald-lp2|donald postscript printer:\
:lp=/dev/lp0:\
:sf:\
:sh:\
:if=/usr/local/lib/psf/psflpd:\
:sd=/var/spool/lpd/donald-lp2:\
:lf=/var/log/lpd-errs
