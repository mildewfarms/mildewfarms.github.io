#!/bin/bash

# Configure SSHD
FILE="/etc/ssh/sshd_config"
echo "Modifying $FILE"
echo ",s/#\?Protocol .*/Protocol 2/
,s/#\?PermitRootLogin .*/PermitRootLogin no/
w
q
" | ed $FILE
