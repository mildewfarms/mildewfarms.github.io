Listing 1 Red Hat AS3 OS config

; rhas3.ini
; Define default OS settings for Red Hat AS 3

[dns]
domain      = "mydomain.com"
nameserver  = "10.0.1.10"

[packages]
resolve_dep = "yes"

[services]
enable      = "ntpd"
disable     = "cups mdmonitor mdmpd"

[scripts]
postinstall = "network services pkg-rpms ssh ntp"

[kickstart_command]
auth        = "--enablemd5 --enableshadow"
network     = "--bootproto dhcp --device eth0"
rootpw      = "S0meP@ssword"
timezone    = "US/Central"
bootloader  = "--location=mbr"
firewall    = "--disabled"
install     = "yes"
keyboard    = "us"
lang        = "en_US"
langsupport = "--default en_US"
mouse       = "generic3ps/2"
reboot      = "yes"
text        = "yes"
zerombr     = "no"

[kickstart_partition]
autopart    = "no"
clearpart   = "--all"

[kickstart_package_add]
gcc
ntp

[kickstart_package_remove]
man-pages-cs
man-pages-da
man-pages-de

[kickstart_post]
cd /tmp
/usr/bin/wget -q -O getpostinstall @@base_url@@/post/getpostinstall
sh /tmp/getpostinstall
rm -f /tmp/getpostinstall
