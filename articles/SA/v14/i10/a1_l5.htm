Listing 5 An example initialization script for starting Samba in the smb project

#!/sbin/sh
#
conf=/usr/local/samba/lib/smb.conf

case "$1" in
'start')
        if [ -f $conf ] ;then
                /usr/bin/newtask -p smb /usr/local/samba/sbin/smbd -D
                /usr/bin/newtask -p smb /usr/local/samba/sbin/nmbd -D
        fi
        ;;

'stop')
        /usr/bin/pkill smbd
        /usr/bin/pkill nmbd
        ;;
*)
        echo "Usage: $0 { start | stop }"
        exit 1
        ;;
esac
exit 0
