Listing 5 Import ZFS filesystems into local zones

35 root@beastie /root
# zonecfg -z zfsTest
zonecfg:zfsTest> add fs
zonecfg:zfsTest:fs> set dir=/u02
zonecfg:zfsTest:fs> set special=pool1/zoneFS
zonecfg:zfsTest:fs> set type=zfs
zonecfg:zfsTest:fs> end
zonecfg:zfsTest> verify
zonecfg:zfsTest> commit
zonecfg:zfsTest> exit
