Listing 4 Function to generate hidden decoy fields

function inputfield($sType="text",$aParams='') {
  $sText="<input type=\"$sType\"
  if ( $aParams!=''  )
    foreach ( $aParams as $key=>$value ) {
      if ( $key=='name' ) {
        $sName=$sValue;
      } else { 
        $sText.=" \"$key\"=\"$value\" ";
      }
    } 
  }
  $nDecoys=rand(10);
  $nReal=rand($nDecoys);
  for ( $i=1;$i++;$i<=$nDecoys ) {
    print $sText;
    if ( $i=$nReal ) {
      print " name=\"$sName_$i\" ";
    } else {
      print " name=\"$sName$i\" ";
    }
    print ">";
  }  
}
