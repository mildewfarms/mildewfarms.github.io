Listing 3 auto_fssnap

#!/usr/bin/bash
#auto_fssnap
#executable automount map
#JRCB 3-dec-02

FSSNAP_DEL=/usr/local/etc/fssnapdel
HOST=`uname -n`

snap_mount()
{
  SLASH=/
  KEY=$1
  #KEY to FS: eg _var_mail to /var/mail
  FS=/snap/${FS//_/$SLASH}
  SNAPDEV=`fssnap -o bs=/share/backingstore/$HOST,unlink $FS`
  echo $SNAPDEV
}

case "$1" in 
  /|/var) /etc/init.d/xntpd stop; snap_mount; /etc/init.d/xntpd start ;;
       *) snap_mount ;;
esac
$FSSNAP_DEL $SNAPDEV &
