Listing 2 An example BIND configuration

    ## named.conf - configuration for bind
    #

    options {
      directory "/var/named";
      pid-file "/var/named/named.pid";
    };

    zone  "." {
      type hint;
      file  "named.ca";
    };

    zone  "example.com" {
      type master;
      file  "example.zone";
      allow-transfer { any; };
      allow-update { 192.168.1.0/24; };
    };
