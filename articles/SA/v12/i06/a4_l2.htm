#!/usr/bin/ksh

#
# add gnut user and group
#
mkgroup -A id='101' gnut
mkuser id='101' pgrp='gnut' \     home='/usr/local/gnut' gnut

#
# mkssys - add gnut as an AIX subsystem
#
echo "adding gnut to SRC... \c"
mkssys -p /usr/local/bin/gnut -u 101 \
    -s gnut -a"-d -c \     /usr/local/etc/gnutrc" \
    -S -n 15 -f 9
if [[ $? = 0 ]]; then
    echo done
    exit 0
else
    echo failed
    exit 1
fi

exit 0
