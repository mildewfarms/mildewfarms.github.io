Listing 4 Build steps

##
## define
##
LPPDIR=$HOME/aixadm.gnut

##
## prep
##
mkdir -p \ $LPPDIR/rte/root/usr/local/gnut/upload
mkdir -p \ $LPPDIR/rte/root/usr/local/gnut/doc
mkdir -p \ $LPPDIR/rte/root/usr/local/gnut/etc
chown -R bin:bin $LPPDIR/rte/root
chown -R gnut:gnut \ $LPPDIR/rte/root/usr/local/gnut/
cp -p data/gnutrc \ $LPPDIR/rte/root/usr/local/gnut/etc/
chown gnut:gnut \ $LPPDIR/rte/root/usr/local/etc/gnutrc

##
## compile
##
cd gnut-0.4.28
./configure
make

##
## install
##
make install DESTDIR=$LPPDIR/rte/root

cp doc/TUTORIAL \ $LPPDIR/rte/root/usr/local/gnut/doc/
cp doc/gnut*.html \ $LPPDIR/rte/root/usr/local/gnut/doc/

##
## package
##

cd $LPPDIR
lppbuild
