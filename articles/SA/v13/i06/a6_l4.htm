Listing 4 setenv_path

# Standard Solaris Pathnames
#

PATH=/sbin:/usr/sbin:/usr/bin:/usr/ccs/bin

MANPATH=/usr/share/man:/usr/dt/share/man:/usr/openwin/share/man:/usr/local/man

#
# Custom Pathnames (Situational)
#

for path in /opt/SUNWspro/bin /usr/mks/mkssi/sol2.4
do
    if [ -d $path ]
    then
        PATH=$PATH:$path
    fi
done

for path in /usr/mks/man /opt/SUNWspro/man /opt/SUNWste/license_tools/man
do
    if [ -d $path ]
    then
        MANPATH=$MANPATH:$path
    fi
done

#
# Custom Pathnames (Standard)
#

PATH=$PATH:/usr/ucb:/usr/sadm/bin:/usr/local/bin:/etc/init.d

PATH=$PATH:/opt/logins/bin:/opt/admin/bin:/opt/admin/sbin:/opt/monitor/ \
  bin:/opt/users/sbin:/opt/users/database/sbin:/opt/changeident/bin:/opt/ \
  autoconf/bin:/opt/pwadmin/bin:/var/opt/build/install:.

MANPATH=$MANPATH:/opt/docs/man

FPATH=/opt/functions

#
# Export Variables
#

export PATH MANPATH FPATH
