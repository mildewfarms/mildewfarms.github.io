Listing 2: show.code

#######################################################
# show.code - display log name used by fchange.sh
#######################################################
# This is an fchange.sh support program.

# make sure this is being run by a posix compliant shell
test "$((1))" = 1 || exec ksh $0 "$@"

# default files file name
: ${fn:=fchange.files}

awk '/^(:|#|$)/ {next}
{print $1}' $fn |
while read fn
do
sum=$(echo $fn | sum -r | awk '{printf("%s_%s\n", $1, $2)}')
echo "$sum  $fn"
done
