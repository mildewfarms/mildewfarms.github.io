/.*/ {
disksize = sprintf("%d", $3);
if (diskused >= 10)
diskused = (disksize * 1.1) / 1024;
else
diskused = (disksize * 1.5) / 1024;
if (diskused >= 1)
printf("%s\t%s\t%s\t%d\n", HOST, $1, $6,
diskused);
else
printf("%s\t%s\t%s\t1\n", HOST, $1, $6);
}
