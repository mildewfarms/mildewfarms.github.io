Listing 2: Parse the QUERY_STRING passed to the server script from the client's browser

#!/bin/sh

eval `echo $QUERY_STRING | sed 's/'"'"'/%27/g' | \
sed 's/;/%27/g' | \
awk 'BEGIN { RS="&"; FS="=" }
$1 ~ /^[a-zA-Z][a-zA-Z0-9_]*$/ {
printf "QS_%s=%c%s%c\n", $1, 39, toupper($2), 39 }' `
