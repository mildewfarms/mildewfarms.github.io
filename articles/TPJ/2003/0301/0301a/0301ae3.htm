add_perl5lib () {
  guts_perl5lib ${1} add
}

del_perl5lib ()
{
  guts_perl5lib ${1} del
}

guts_perl5lib ()
{
  for i in $(find ${1}/lib -name ${PERL_CURRENT_VERSION} -type d | \
 sort -r)
  do
    ${2}path -f -p PERL5LIB $(dirname $i)
  done
  export PERL5LIB
}
